{\Large Ejercicios de repaso de polinomios.}
\begin{enumerate}
    \item Dados los polinomios: 
    
        $A(x) = x^4-2x^2+2x-3$  \hspace{1cm} $B(x) = 3x^3+2x-1$ \hspace{1cm} $C(x)=4x^4-x^3+8x$
        
        Calcula:
        \begin{enumerate}
            \item $2A(x) - B(x) + 3C(x)$
            \item $A(x)+3B(x)-2C(x)$
        \end{enumerate}
    \item Dados los polinomios:
    
        $A(x)=3x^3-x+4$ \hspace{1cm} $B(x)=5x^4-2x^3+2$

        Calcula:
        \begin{enumerate}
            \item $2A(x)+3B(x)$
            \item $A(x) \cdot B(x)$
        \end{enumerate}

    \item Saca factor común en las siguientes expresiones.
        \begin{enumerate}
            \item $3b+12$
            \item $12xy+30z$
            \item $9x^2y+21x$
            \item $7ab-14ac+21ad$
            \item $7x-21$
            \item $12xy-30xz$
            \item $4u^2v^2-12uv^2$
            \item $12abc^2-4^2bc+6ab^2c$
        \end{enumerate}

    \item Desarrolla los siguientes cuadrados sin hacer la multiplicación:
    \begin{enumerate}
        \item $(x+6)^2$
        \item $(2x+6y)^2$
        \item $(a^2-2)^2$
        \item $(2x-6)^2$
    \end{enumerate}
    \item Completa las siguientes expresiones par que sean cuadrados perfectos:
        \begin{enumerate}
            \item $16x^2+24xy+.......=(..............)^2$
            \item $9x^4-.......+16y^6=(..............)^2$
            \item $.......-6x^4y^6+y^12=(..............)^2$
            \item $a^2+.......+b^4=(..............)^2$
        \end{enumerate}
    \item Efectúa las siguientes divisiones:
        \begin{enumerate}
            \item $(x^4-5x^2+3x-4):(x^2-x+2)$
            \item $(x^6+6x^5+8x^3-5x+2):(x+2)$
            \item $(x^8-256):(x+2)$
        \end{enumerate}
\end{enumerate}
